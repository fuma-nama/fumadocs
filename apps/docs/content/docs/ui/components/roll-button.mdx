## Usage

Add this component to your page.

```mdx title="app/docs/[[...slug]]/page.tsx"
import { RollButton } from 'next-docs-ui/components'

<DocsPage toc={toc} tree={tree}>
  <RollButton />
  <MDXContent>
    <h1>{page.title}</h1>
    <MDX />
  </MDXContent>
</DocsPage>
```
